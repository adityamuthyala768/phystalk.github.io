--- simulations.js --- // Electric Field Visualizer // Canvas-based physics simulation class Charge { constructor(x, y, magnitude = 1) { this.x = x; this.y = y; this.magnitude = magnitude; this.id = Math.random().toString(36).substr(2, 9); } setMagnitude(val) { this.magnitude = Math.max(0.1, Math.min(3, val)); } resetPosition(x, y) { this.x = x; this.y = y; } } class ElectricFieldSimulator { constructor(canvasId) { this.canvas = document.getElementById(canvasId); this.ctx = this.canvas.getContext('2d'); this.charges = []; this.maxCharges = parseInt(document.getElementById('maxCharges')?.value || 5); this.arrowScale = parseInt(document.getElementById('arrowScale')?.value || 2); this.showArrows = document.getElementById('showArrows')?.checked || true; this.showLines = document.getElementById('showLines')?.checked || true; this.showDivergence = document.getElementById('showDivergence')?.checked || false; this.dragging = null; this.init(); } init() { this.resize(); this.setupEventListeners(); this.updateChargeDisplay(); this.render(); } resize() { const parent = this.canvas.parentElement; this.canvas.width = parent.clientWidth; this.canvas.height = parent.clientHeight; this.render(); } setupEventListeners() { window.addEventListener('resize', () => this.resize()); this.canvas.addEventListener('click', (e) => this.handleClick(e)); this.canvas.addEventListener('contextmenu', (e) => e.preventDefault()); // Controls document.getElementById('addChargeBtn')?.addEventListener('click', () => this.addCharge()); document.getElementById('clearChargesBtn')?.addEventListener('click', () => this.clearCharges()); document.getElementById('maxCharges')?.addEventListener('change', (e) => { this.maxCharges = parseInt(e.target.value); if (this.charges.length > this.maxCharges) { this.removeCharge(this.charges.length - 1); } }); document.getElementById('chargeMag')?.addEventListener('input', (e) => { const val = parseFloat(e.target.value); document.getElementById('chargeMagValue')?.textContent = val.toFixed(1); }); document.getElementById('arrowScale')?.addEventListener('input', (e) => { const val = parseFloat(e.target.value); this.arrowScale = Math.round(val); document.getElementById('arrowScaleValue')?.textContent = this.arrowScale; }); document.getElementById('showArrows')?.addEventListener('change', (e) => { this.showArrows = e.target.checked; this.render(); }); document.getElementById('showLines')?.addEventListener('change', (e) => { this.showLines = e.target.checked; this.render(); }); document.getElementById('showDivergence')?.addEventListener('change', (e) => { this.showDivergence = e.target.checked; this.render(); }); } addCharge(x, y) { const centerX = this.canvas.width / 2; const centerY = this.canvas.height / 2; const mag = parseFloat(document.getElementById('chargeMag').value) || 1; // Place in available positions around center const positions = [ [0.8, 0.5], [1.2, 0.5], [0.8, 1.5], [1.2, 1.5], [0.5, 0.8], [0.5, 1.2], [1.5, 0.8], [1.5, 1.2], [0.3, 0.5], [1.7, 0.5], [0.3, 1.5], [1.7, 1.5], [0.5, 0.3], [0.5, 1.7], [1.5, 0.3], [1.5, 1.7] ]; const pos = x !== null ? [x, y] : positions[Math.floor(Math.random() * positions.length)]; const cx = centerX + pos[0] * 80; const cy = centerY + pos[1] * 80; this.charges.push(new Charge(cx, cy, mag)); this.render(); } clearCharges() { this.charges = []; this.render(); } removeCharge(index) { this.charges.splice(index, 1); this.render(); } handleClick(e) { const rect = this.canvas.getBoundingClientRect(); const x = e.clientX - rect.left; const y = e.clientY - rect.top; // Check if clicked on a charge const clickIndex = this.findChargeAt(x, y); if (clickIndex !== -1) { this.dragging = this.charges[clickIndex]; this.canvas.style.cursor = 'move'; } else { this.dragging = null; this.canvas.style.cursor = 'default'; } } findChargeAt(x, y) { const radius = 15; for (let i = 0; i < this.charges.length; i++) { const dx = this.charges[i].x - x; const dy = this.charges[i].y - y; if (Math.sqrt(dx*dx + dy*dy) < radius) return i; } return -1; } render() { this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height); // Calculate field if showing arrows if (this.showArrows && this.charges.length > 0) { this.calculateAndDrawField(); } // Draw field lines if (this.showLines) { this.drawFieldLines(); } // Draw divergence heatmap if (this.showDivergence) { this.drawDivergenceHeatmap(); } // Draw charges for (const charge of this.charges) { this.drawCharge(charge); } } calculateAndDrawField() { const k = 9e9; // Coulomb constant (scaled) const dxMin = 10; const dyMin = 10; const maxArrows = 50; const stepX = Math.min(dxMin, this.canvas.width / (maxArrows + 1)); const stepY = Math.min(dyMin, this.canvas.height / (maxArrows + 1)); let arrowCount = 0; for (let x = stepX; x < this.canvas.width - stepX; x += stepX) { for (let y = stepY; y < this.canvas.height - stepY; y += stepY) { if (arrowCount >= maxArrows) break; // Calculate field at point let Ex = 0, Ey = 0; for (const charge of this.charges) { const dx = x - charge.x; const dy = y - charge.y; const rSq = dx*dx + dy*dy; if (rSq > 100) { const r = Math.sqrt(rSq); const factor = k * charge.magnitude / rSq; Ex += factor * (dx / r); Ey += factor * (dy / r); } } // Draw arrow const arrowX = x - 20 + (Math.abs(Ex) * 20); const arrowY = y - 20 + (Math.abs(Ey) * 20); const arrowSize = Math.max(5, Math.min(30, Math.sqrt(Ex*Ex + Ey*Ey) / 1000 * 100)); this.ctx.beginPath(); this.ctx.strokeStyle = `rgba(255, ${200 - Math.abs(Ex)}, ${100 - Math.abs(Ey)}, 0.8)`; this.ctx.lineWidth = 1.5; this.ctx.moveTo(arrowX, arrowY); this.ctx.lineTo(arrowX - Ex * 20, arrowY - Ey * 20); this.ctx.stroke(); arrowCount++; } } } drawFieldLines() { const nLines = Math.min(3, this.charges.length); const lines = []; for (let i = 0; i < nLines; i++) { const startX = Math.random() * this.canvas.width; const startY = Math.random() * this.canvas.height; lines.push({ x: startX, y: startY, visited: new Set() }); } const iterations = 30; const stepSize = 5; for (const line of lines) { for (let iter = 0; iter < iterations; iter++) { if (line.x > this.canvas.width || line.x < 0 || line.y > this.canvas.height || line.y < 0) break; const chargeIdx = this.findChargeAt(line.x, line.y); if (chargeIdx !== -1) { line.x = this.charges[chargeIdx].x; line.y = this.charges[chargeIdx].y; line.visited.clear(); } } } // Draw all lines const color = (line) => { const count = line.visited.size; if (count > 0) { return `hsl(${(count % 360)}, 70%, ${60 - Math.min(60, count / 5)}%)`; } return '#888'; }; this.ctx.strokeStyle = color(lines[0]); this.ctx.lineWidth = 1; for (const line of lines) { this.ctx.beginPath(); this.ctx.moveTo(line.x, line.y); for (let i = 0; i < 30; i++) { const nextX = line.x; const nextY = line.y; this.ctx.lineTo(nextX, nextY); // Move along field (simplified tracing) const chargeIdx = this.findChargeAt(nextX, nextY); if (chargeIdx !== -1) { line.visited.add(nextX + ',' + nextY); const angle = -Math.PI + Math.atan2(line.y - this.charges[chargeIdx].y, nextX - this.charges[chargeIdx].x); line.x = nextX + Math.cos(angle) * stepSize; line.y = nextY + Math.sin(angle) * stepSize; } else { line.visited.add(nextX + ',' + nextY); const angle = Math.atan2(line.y - line.y, nextX - nextX); line.x = nextX + Math.cos(angle) * stepSize; line.y = nextY + Math.sin(angle) * stepSize; } } this.ctx.stroke(); } } drawDivergenceHeatmap() { const imageData = this.ctx.getImageData(0, 0, this.canvas.width, this.canvas.height); const data = imageData.data; const k = 9e9; for (let j = 0; j < this.canvas.height; j++) { for (let i = 0; i < this.canvas.width; i++) { let div = 0; for (const charge of this.charges) { const dx = i - charge.x; const dy = j - charge.y; const rSq = dx*dx + dy*dy; if (rSq > 100) { const r = Math.sqrt(rSq); div += -k * charge.magnitude / rSq; // Divergence } } // Map divergence to color let color; if (div > 0) { color = `rgb(255, ${Math.min(255, Math.floor(div * 100))}, 0)`; } else { color = `rgb(${Math.max(0, Math.floor(-div * 100))}, 255, 0)`; } const idx = (j * this.canvas.width + i) * 4; data[idx] = parseInt(color.split(',')[0]); data[idx+1] = parseInt(color.split(',')[1]); data[idx+2] = parseInt(color.split(',')[2]); data[idx+3] = 255; }
